#include <iostream>

int main()
{
    double altura, largura, comprimento, volume;

    std::cout << "Entre com a altura, argura e o comprimento.\n>>>> ";
    std:: cin >> altura >> largura >> comprimento ;
   
    volume = altura * largura * comprimento;

    std::cout << "\n======================================\n";
    std::cout << "altura: " << altura << " cm\n"
              << "largura: " <<largura << " cm\n"
              << "comprimento: " << comprimento << " cm\n";
    std::cout << "\nO volume e: " << volume << " cm cubicos\n";

}
